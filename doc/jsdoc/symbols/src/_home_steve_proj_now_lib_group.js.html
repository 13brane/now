1 /**
  2  * @fileOverview This file contains the Group class, which can be used
  3    to simultaneously manipulate some subset of all connected users by
  4    calling functions or modifying their local now namespaces.
  5  */

var EventEmitter = require('events').EventEmitter;
  8 var nowUtil = require('./nowUtil').nowUtil;
  9 var Proxy = require('./proxy');
 10 var ScopeTable = require('./scopeTable').ScopeTable;
 11 
 12 exports.initialize = function (nowjs) {
 13   var fn = require('./function').init(nowjs);
 14 
 15   /**
 16    * @name Group
 17 
 18    * @class Represents a group containing some subset of all
 19      users. Each group has its own {@link Group#now} namespace.
 20 
 21    * @property {String} groupName The name associated with the group.
 22    */
 23 
 24   var Group = function (groupName) {
 25     //all users in the group
 26     this.users = {};
 27 
 28     this.groupName = groupName;
 29 
 30     this.excludes = {};
 31     /**
 32      * @name Group#now
 33      * @namespace The now namespace for this particular group. Actions
 34      * to this namespace affect all users that are members of the
 35      * group. All functions that are called will be called in the
 36      * context of each individual user in the group.
 37      */
 38     this.now = Proxy.wrap(this);
 39     //group scope table
 40     this.scopeTable = new ScopeTable();
 41   };
 42 
 43   Group.prototype.__proto__ = EventEmitter.prototype;
 44 
 45   /**
 46    * @name count
 47    * @function
 48    * @memberOf Group#
 49 
 50    * @description Used to find the cardinality of the group (how
 51    * many users it contains).
 52 
 53    * @param {Function} callback Called with a Number corresponding to
 54    * the group's user count.
 55    */
 56   Group.prototype.count = function (callback) {
 57     callback(Object.keys(this.users).length);
 58   };
 59 
 60   /**
 61    * @name getUsers
 62    * @function
 63    * @memberOf Group#
 64 
 65    * @description Used to retrieve a list of the client IDs
 66    * corresponding to all users in the group.
 67 
 68    * @param {Function} callback Called with an Array of Strings
 69    * corresponding to the client IDs of all users in the group.
 70    */
 71   Group.prototype.getUsers = function (callback) {
 72     callback(Object.keys(this.users));
 73   };
 74 
 75   /**
 76    * @name connected
 77    * @function
 78    * @memberOf Group#
 79    * @deprecated As of 0.7.0. Use Now:connect instead.
 80    */
 81   Group.prototype.connected = function (callback) {
 82     nowjs.on('connect', callback);
 83   };
 84 
 85   /**
 86    * @name disconnected
 87    * @function
 88    * @memberOf Group#
 89    * @deprecated As of 0.7.0. Use Now:disconnect instead.
 90    */
 91   Group.prototype.disconnected = function (callback) {
 92     nowjs.on('disconnect', callback);
 93   };
 94 
 95   /**
 96    * @name addUser
 97    * @function
 98    * @memberOf Group#
 99    * @description Adds the user identified by clientId to this group.
100 
101    * @param {String} clientId The client ID associated with the target
102    * user.
103    */
104   Group.prototype.addUser = function (clientId) {
105     var self = this;
106     this.hasClient(clientId, function (hasClient) {
107       if (hasClient) {
108         return;
109       }
110       nowjs.getClient(clientId, function () {
111         // Scoping note: `self` refers to the group, `this` refers to
112         // the new client.
113         self.users[this.user.clientId] = this;
114         this.groups[self.groupName] = self;
115 
116         // Send the client any new variables that it does not already have
117         var toSend = {};
118         var keys = Object.keys(self.scopeTable.data);
119         var key, val;
120         for (var i = 0, ll = keys.length; i < ll; i++) {
121           key = keys[i];
122           val = self.scopeTable.get(key);
123           if (val !== nowUtil.noop && !Array.isArray(val) &&
124               !nowUtil.hasProperty(this.scopeTable.data, key)) {
125             toSend[key] = nowUtil.getValOrFqn(val, key);
126           }
127         }
128 
129         if (self.isSuperGroup || !nowUtil.isEmptyObj(toSend)) {
130           this.socket.emit('rv', toSend);
131         }
132         var user = nowUtil.clone(this, {'_events': self._events});
133 
134         /**
135          * @name Group#join
136          * @event
137          * @description Called in the context of a user who has just been
138          * added to the group.
139          */
140         self.emit.apply(user, ['join']);
141       });
142     });
143   };
144 
145   /**
146    * @name removeUser
147    * @function
148    * @memberOf Group#
149    * @version 0.7.0
150    * @description Removes the user identified by clientId from this group.
151 
152    * @param {String} clientId The client ID associated with the target
153    * user.
154    */
155   Group.prototype.removeUser = function (clientId) {
156     var self = this;
157     this.hasClient(clientId, function (hasClient) {
158       if (!hasClient) {
159         return;
160       }
161       var user = nowUtil.clone(self.users[clientId], {'_events': self._events});
162 
163       /**
164        * @name Group#leave
165        * @event
166        * @version 0.7.0
167        * @description Called in the context of a user who has just been
168        * removed from the group.
169        */
170       self.emit.apply(user, ['leave']);
171       // Delete all remote functions that are part of this group from
172       // the user.
173       var fqns = Object.keys(self.scopeTable.data);
174       for (var i = 0; i < fqns.length; i++) {
175         if (typeof self.scopeTable.data[fqns[i]] === 'function' &&
176             self.scopeTable.data[fqns[i]] !== nowUtil.noop &&
177             user.scopeTable.data[fqns[i]] === undefined) {
178           // Tell the user to delete his function.
179           user.deleteVar(fqns[i]);
180         }
181       }
182       delete self.users[clientId];
183       delete user.groups[self.groupName];
184     });
185   };
186 
187   /**
188    * @memberOf Group#
189    * @function
190    * @name exclude
191    * @version 0.7.0
192 
193    * @description Returns a new Group that is identical to the calling
194    * group, except with the specified clients excluded. The returned
195    * group automatically updates to accommodate any changes made to
196    * its parent group.
197 
198    * @param {Array} clientIds A list of client IDs corresponding to
199    * clients to exclude.
200 
201    * @type Group
202    */
203   Group.prototype.exclude = function (clientIds) {
204     var excludes = {};
205     for (var i = 0; i < clientIds.length; i++) {
206       excludes[clientIds[i]] = true;
207     }
208     var group = nowUtil.clone(this, {excludes: nowUtil.clone(this.excludes, {excludes: excludes})});
209     group.now = Proxy.wrap(group);
210     return group;
211   };
212 
213   /**
214    * @memberOf Group#
215    * @function
216    * @name hasClient
217 
218    * @description Used to determine a given user's membership in the
219    * group.
220 
221    * @param {String} clientId The client ID associated with the target
222    * user.
223 
224    * @param {Function} callback Called with a Boolean indicating the
225    * user's membership in the group.
226    */
227   //returns boolean whether or not user is in a group
228   Group.prototype.hasClient = function (clientId, callback) {
229     callback(this.users[clientId] !== undefined);
230   };
231 
232   Group.prototype.get = function (fqn) {
233     var value = this.scopeTable.get(fqn);
234     // If this is a regular group, look in `everyone` for the value
235     if (value === undefined && !this.isSuperGroup) {
236       value = nowjs.getGroup('everyone').scopeTable.get(fqn);
237     }
238 
239     // If this is a function, return a multicaller.
240     if (typeof value === 'function') {
241       var group = nowUtil.clone(this, {fqn: fqn});
242       value = fn.multicall.bind(group);
243     }
244     return value;
245   };
246 
247   Group.prototype.deleteVar = function (fqn) {
248     var obj = nowUtil.clone(this, {fqn: fqn});
249     nowjs.emit('groupdel', obj);
250   };
251 
252   Group.prototype.set = function (fqn, val) {
253     var obj = nowUtil.clone(this, {fqn: fqn, val: val});
254     nowjs.emit('grouprv', obj);
255   };
256 
257   return Group;
258 };
